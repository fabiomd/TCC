;;; ###  follow goa install guide

;; install parser-js in ~/quicklisp/local-projects

;; check http://marijnhaverbeke.nl/parse-js/ for details

git clone https://github.com/marijnh/parse-js.git

;;or use 

ln -s $(pwd) ~/quicklisp/local-projects/
in your parser-js folder

;;; ###  install webassmebly

;; check http://webassembly.org/getting-started/developers-guide/

git clone https://github.com/juj/emsdk.git

cd emsdk

./emsdk install sdk-incoming-64bit binaryen-master-64bit

./emsdk activate sdk-incoming-64bit binaryen-master-64bit

./emsdk install latest
./emsdk update latest

source ./emsdk_env.sh

;;; now we can compile a program

;; the program need to have the .html extension

;; like emcc hello.c -s WASM=1 -o hello.html

;; run emcc in port

emrun --no_browser --port 8080 .
emrun --browser firefox hello.html
